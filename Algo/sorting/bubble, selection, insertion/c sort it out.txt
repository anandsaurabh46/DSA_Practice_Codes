#include <iostream>
#include <vector>

using namespace std;

void swap(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int main() {
	int n;
	cin >> n;
	vector<int> arr(n, 0);
	vector<int> index(n, 0);
	for(int i=0; i<n; i++) {cin >> arr[i]; index[i]=i;}
	for(int i=0; i<n-1; i++) {
        bool flag = true;
        for(int j=0; j<n-i-1; j++) {
            if(arr[j]>arr[j+1]) {
                flag = false;
                swap(arr[j], arr[j+1]);
                swap(index[j], index[j+1]);
            }
        }
        if(flag) break;
    }
    for(int i=0; i<n; i++) cout << index[i] << ' ';
}
